 37 /**
 38  * @fileoverview This small file provides conveniences wrappers around legacy
 39  * address book interfaces.
 40  * @author Jonathan Protzenko
 41  */
 42 
 43 var EXPORTED_SYMBOLS = [
 44   'kPersonalAddressBookUri', 'kCollectedAddressBookUri',
 45   'getAddressBookFromUri', 'saveEmailInAddressBook'
 46 ];
 47 
 48 const Ci = Components.interfaces;
 49 const Cc = Components.classes;
 50 const Cu = Components.utils;
 51 const Cr = Components.results;
 52 
 53 const rdfService = Cc["@mozilla.org/rdf/rdf-service;1"]
 54                    .getService(Ci.nsIRDFService);
 55 
 56 const kPersonalAddressBookUri = "moz-abmdbdirectory://abook.mab";
 57 const kCollectedAddressBookUri = "moz-abmdbdirectory://history.mab";
 58 
 59 /**
 60  * Get one of the predefined address books through their URI.
 61  * @param aUri the URI of the address book, use one of the consts above
 62  * @return nsIAbDirectory the address book
 63  */
 64 function getAddressBookFromUri(aUri) {
 65   let r = rdfService.GetResource(aUri)
 66     .QueryInterface(Ci.nsIAbDirectory);
 67   return r;
 68 }
 69 
 70 /**
 71  * Just a one-liner to add an email to the given address book, without any extra
 72  *  properties.
 73  * @param aBook the nsIAbDirectory
 74  * @param aEmail the email
 75  * @param aName (optional) the name
 76  * @return nsIAbCard
 77  */
 78 function saveEmailInAddressBook(aBook, aEmail, aName) {
 79   let card = Cc["@mozilla.org/addressbook/cardproperty;1"]  
 80              .createInstance(Ci.nsIAbCard);
 81   //card.setProperty("FirstName", "John");
 82   //card.setProperty("LastName", "Smith");
 83   card.displayName = aName;
 84   card.primaryEmail = aEmail;
 85   card.setProperty("AllowRemoteContent", true);
 86   return aBook.addCard(card);
 87 }
